GFW

SSR

Install

git clone -b manyuser https://github.com/Raikyou/shadowsocksr
cd shadowsocksr
bash initcfg.sh
vi user-config.json
cd shadowsocks
./logrun.sh

Config

// single user
"server_port": 8388,
"password": "m",
"method": "rc4-md5",
"protocol": "auth_aes128_md5",
"obfs": "tls1.2_ticket_auth",

// only ports and password
"port_password":{
        "80":"password1",
        "443":"password2"
        },

 // all paras   
"port_password":{
        "8388":{"protocol":"auth_simple", "password":"abc", "obfs":"plain", "obfs_param":""},
        "8389":{"protocol":"origin", "password":"abcde"}
        },

Startup

chmod +x /etc/rc.local
vi etc/rc.local
# add '/bin/bash /root/shadowsocksr/shadowsocks/logrun.sh'

Update

cd shadowsocksr
git pull
# restart after succeeding
