#set：一组key的集合，无序，只显示包含的无重复元素
s1 = set([1, 1, 2, 2, 3, 3])
print(s1)                   #输出{1, 2, 3}
s2 = set([2, 3, 4])
print(s1 & s2)               #交集{2, 3}
print(s1 | s2)              #并集{1, 2, 3, 4}

def calc(*numbers):    #可变参数，调用为calc(1, 2)；若定义为calc(numbers)，则调用为calc([1, 2])
